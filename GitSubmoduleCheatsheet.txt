# 1Ô∏è‚É£ Clone a repo *with* all its submodules
git clone https://github.com/USERNAME/portfolio4.git
cd portfolio4
git submodule update --init --recursive

# 2Ô∏è‚É£ Add a new submodule (and include it in the parent)
# Example: add a new project folder as a submodule
git submodule add https://github.com/USERNAME/NewProject.git NewProject

# This creates an entry in .gitmodules and links it to your main repo.
# Now commit both the new folder and the .gitmodules changes:
git add .gitmodules NewProject
git commit -m "Add NewProject as submodule"
git push

# ‚úÖ Tip: When someone else clones the main repo later, they just need:
# git clone https://github.com/USERNAME/portfolio4.git
# git submodule update --init --recursive

# 3Ô∏è‚É£ Work on a submodule
cd Pokedex
# (make your edits)
git add .
git commit -m "Update Pokedex feature"
git push

# Then update the main repo pointer
cd ..
git add Pokedex
git commit -m "Update Pokedex submodule pointer"
git push

# 4Ô∏è‚É£ Pull the latest updates (main + submodules)
git pull
git submodule update --remote --merge

# 5Ô∏è‚É£ Initialize submodules after cloning (if not yet done)
git submodule update --init --recursive

# 6Ô∏è‚É£ See submodule status
git submodule status

# 7Ô∏è‚É£ Remove a submodule
git submodule deinit -f path/to/module
rm -rf .git/modules/path/to/module
git rm -f path/to/module
git commit -m "Remove submodule path/to/module"
git push

# ==============================
# üí° TIPS
# - Always commit and push inside submodules first.
# - Then update the main repo‚Äôs pointer afterward.
# - Avoid nesting submodules within other submodules.

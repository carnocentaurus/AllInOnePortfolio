# 1Ô∏è‚É£ Clone a repo *with* all its submodules
git clone https://github.com/USERNAME/portfolio4.git
cd portfolio4
git submodule update --init --recursive

# 2Ô∏è‚É£ Add a new submodule (3 ways)
# --------------------------------

# üÖê Create a brand-new submodule entirely via CLI:
gh repo create USERNAME/NewProject --public --confirm   # or --private
mkdir NewProject && cd NewProject
git init
echo "# NewProject" > README.md
git add README.md
git commit -m "Initial commit"
git branch -M main
git remote add origin https://github.com/USERNAME/NewProject.git
git push -u origin main
cd ../portfolio4
git submodule add https://github.com/USERNAME/NewProject.git NewProject
git add .gitmodules NewProject
git commit -m "Add NewProject as submodule"
git push

# üÖë If you already have a local folder you built manually (not a repo yet):
cd NewProject
git init
git add .
git commit -m "Initial commit"
gh repo create USERNAME/NewProject --public --confirm
git branch -M main
git remote add origin https://github.com/USERNAME/NewProject.git
git push -u origin main
cd ../portfolio4
rm -rf NewProject                     # remove the untracked folder
git submodule add https://github.com/USERNAME/NewProject.git NewProject
git add .gitmodules NewProject
git commit -m "Convert NewProject folder into submodule"
git push

# üÖí If the submodule repo already exists elsewhere (you created it manually on GitHub):
cd portfolio4
git submodule add https://github.com/USERNAME/NewProject.git NewProject
git add .gitmodules NewProject
git commit -m "Add existing NewProject repo as submodule"
git push

# ‚úÖ After adding, verify:
git submodule status
cat .gitmodules

# 3Ô∏è‚É£ Work on a submodule
cd Pokedex
# (make your edits)
git add .
git commit -m "Update Pokedex feature"
git push

# Then update the main repo pointer
cd ..
git add Pokedex
git commit -m "Update Pokedex submodule pointer"
git push

# 4Ô∏è‚É£ Pull the latest updates (main + submodules)
git pull
git submodule update --remote --merge

# 5Ô∏è‚É£ Initialize submodules after cloning (if not yet done)
git submodule update --init --recursive

# 6Ô∏è‚É£ See submodule status
git submodule status

# 7Ô∏è‚É£ Remove a submodule
git submodule deinit -f path/to/module
rm -rf .git/modules/path/to/module
git rm -f path/to/module
git commit -m "Remove submodule path/to/module"
git push

# ==============================
# üí° TIPS
# - Always commit and push inside submodules first.
# - Then update the main repo‚Äôs pointer afterward.
# - Avoid nesting submodules within other submodules.
# - Use `--recursive` when cloning or updating to include all submodules.
